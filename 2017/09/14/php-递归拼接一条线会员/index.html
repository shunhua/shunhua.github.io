php 递归拼接一条线会员

2017-09-14

php 递归拼接一条线会员：

  • PHP使用递归&参数引用

public static function parentIdLine($adminid,&$parentAdmin_strs=’’)
{

       $parent_id = self::find($adminid)->parent_id;
       // 构造上级限制
       $parentids = array(0,1); //主要是限制管理员不拿提成          
       if (!in_array($parent_id, $parentids) ) {
          $parentAdmin_strs .= $parent_id.',';
          self::parentIdLine($parent_id,$parentAdmin_strs); 
       } 

     return $parentAdmin_strs;

}
